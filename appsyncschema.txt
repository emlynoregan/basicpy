schema {
	query: Query
}

type Query {
	listPersons(limit: Int, nextToken: String): personPage
}

type personPage {
	items: [person]
	nextToken: String
}

type person {
    pkey: Int!
    created: Int!
    updated: Int!
    enabled: Boolean!
    authid: String
    displayname: String!
    contacts: [ person! ]!
}
